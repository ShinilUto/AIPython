def get_top_amounts(group, key, n=5):
    totals = group.groupby(key)['contb_receipt_amt'].sum()

    # totals를 오름차순으로 정렬 후 마지막 n개 슬라이싱
    return totals.sort_values(ascending=False)[:n]
